=>LIMIT Clause
LIMIT Syntax

METHOD 1
SELECT * FROM tablename
WHRER condition
LIMIT number;

METHOD 2
SELECT column1,column2,.....
FROM tablename
WHERE condition
LIMIT number;

________________________________________________

EXERCIES

SELECT * FROM employees;

SELECT * FROM employees LIMIT 5;

SELECT name,age,salary,city
FROM employees
WHERE salary > 400000
LIMIT 2;

SELECT name,age,salary,city
FROM employees
WHERE salary > 400000
ORDER BY salary;

SELECT name,age,salary,city
FROM employees
WHERE salary > 400000
ORDER BY salary LIMIT 2;

SELECT name,age,salary,city
FROM employees
WHERE salary > 400000
ORDER BY name LIMIT 3;
